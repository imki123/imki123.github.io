JS에서 this란?

1. 기본적으로 this는 window다.

function 으로 생성하든 콜백함수든 어디서든 window다.

console.log(this) //window
function(){
	console.log(this) //window
}


2. 객체메서드, new 생성자, 화살표 함수, bind, call, apply 에서는 바뀐다.

//객체메서드
const counter = {
	value : 0
	inc : function(){ // 여기서 this는 counter이다
	this.value ++
	}
}



//new 생성자
function Person(){  
	this.name = '기영'
}
Person.prototype.hello = function(){
  console.log(this.name) //여기서 this는 Person의 인스턴스이다.
}
const person = new Person()
console.log(person.name) //기영



function hello(a, b, c) {
  name = "기영"
  console.log(`hello ${this.name} ${a} ${b} ${c}`)
}
const obj = {
  name: '임기123'
}
hello() //hello 기영 undefined undefined undefined
//bind
objHello = hello.bind(obj) //hello함수의 this가 obj인 새로운 함수를 리턴한다.
objHello() //hello 임기123 undefined undefined undefined //this가 obj인 hello함수를 실행한다. 
//call, apply
hello.call(obj, 1, 2, 3) //hello 임기123 1 2 3 //hello함수를 실행하는데 this는 obj를 가리킨다. 파라미터로 인수들을 받는다.
hello.apply(obj, [1, 2, 3]) //hello 임기123 1 2 3 //위와 같다. 파라미터로 배열 하나를 받는다.


3. 예외적으로 onclick 등의 이벤트리스너나 jquery등의 라이브러리에서도 바뀔수있다.

button.onclick = function(){
	console.log(this) //여기서 this는 button
}
